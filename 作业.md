#include <stdio.h>

int main() {
	char c,res;
	scanf("%c", &c);
	//输入小写字母转换为大写字母
	if (c >= 'a' && c <= 'z') {
		res = c-'a'+'A';
		printf("%c", res);
	}
	//输入大写字母直接输出
	else if (c >= 'A' && c <= 'Z') {
		printf("%c", c);
	}
	//输入的数字
	else if (c >= '0' && c <= '9') {
		int ans = c - '0';
		printf("%d", ans);
	}
	//其他
	else {
		printf("Other character");
	}
	return 0;
}
