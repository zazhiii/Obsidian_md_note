import java.util.Scanner;  
  
public class Main {  
    public static void main(String[] args) {  
        Scanner s = new Scanner(System.in);  
        int n = s.nextInt();  
        //用数组装输入的数  
        int[] arr = new int[n];  
        for (int i = 0; i < n; i++) {  
            arr[i] = s.nextInt();  
        }  
        //遍历数组查找质数  
        for (int i = 0; i < n; i++) {  
            //是质数则打印输出  
            if (isPrime(arr[i])) {  
                System.out.print(arr[i] + " ");  
            }  
        }  
  
  
    }  
  
    //判断是否为质数  
    private static boolean isPrime(int i) {  
        if (i==1){  
            return false;  
        }  
        //如果为2返回true  
        if (i == 2) {  
            return true;  
        }  
        //大于2  
        
        //j可以小于根号i
        
        for (int j = 2; j <= i / 2; j++) {  
            //有因数则不是质数  
            if (i % j == 0) {  
                return false;  
            }  
        }  
        //循环走完没有因数则为质数 返回true  
        return true;  
    }  
}
